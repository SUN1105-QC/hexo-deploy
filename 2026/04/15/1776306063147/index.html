ツールランチャー HTA ツールセット (基于 HTA 的通用工具启动器及配置管理方案)

发表于 2026-04-16 | 标签： .hta, HTA, 基于

简短摘要 (Summary / Description)
摘要： 本项目是一个基于 HTML 应用程序（HTA）开发的轻量化工具启动框架。
通过 OpsLauncher.hta 提供卡片式交互界面，支持一键运行 .bat 脚本、打开特定路径下的 Excel 或文件夹；
配套 AddTool.hta 实现动态可视化条目添加，并通过 tools.ini 进行持久化数据管理。
无需安装环境，双击即用，适用于 Windows 系统的办公自动化管理。

<script language="VBScript">
Option Explicit

Const BAT_NAME = "起動.bat"
Const TOOLS_FILE = "tools.ini"

Dim shell, fso, baseDir, tools
Set shell = CreateObject("WScript.Shell")
Set fso   = CreateObject("Scripting.FileSystemObject")
Set tools = CreateObject("Scripting.Dictionary")

Function UrlDecode(s)
  Dim i, ch, hex
  UrlDecode = ""
  i = 1
  Do While i <= Len(s)
    ch = Mid(s, i, 1)
    If ch = "%" And i + 2 <= Len(s) Then
      hex = Mid(s, i+1, 2)
      UrlDecode = UrlDecode & Chr(CLng("&H" & hex))
      i = i + 3
    Else
      UrlDecode = UrlDecode & ch
      i = i + 1
    End If
  Loop
End Function

Function GetBaseDir()
  Dim href, p, posQ, posHash, posLastSlash
  href = document.location.href
  href = UrlDecode(href)
  posQ = InStr(href, "?")
  If posQ > 0 Then href = Left(href, posQ - 1)
  posHash = InStr(href, "#")
  If posHash > 0 Then href = Left(href, posHash - 1)
  If LCase(Left(href, 8)) = "file:///" Then
    p = Mid(href, 9)
  ElseIf LCase(Left(href, 7)) = "file://" Then
    p = Mid(href, 8)
  Else
    p = href
  End If
  p = Replace(p, "/", "\")
  posLastSlash = InStrRev(p, "\")
  If posLastSlash > 0 Then
    GetBaseDir = Left(p, posLastSlash - 1)
  Else
    GetBaseDir = p
  End If
End Function

Sub LogError(msg)
  Dim logFile, ts
  On Error Resume Next
  logFile = fso.BuildPath(baseDir, "ErrorLog.txt")
  Set ts = fso.OpenTextFile(logFile, 8, True, -1)
  ts.WriteLine Now & " - " & msg
  ts.Close
  On Error GoTo 0
End Sub

Function IniEscape(s)
  IniEscape = Replace(Replace(Replace(s, vbCr, " "), vbLf, " "), "|", "¦")
End Function

Function IniUnescape(s)
  IniUnescape = Replace(s, "¦", "|")
End Function

Function HtmlEncode(s)
  s = Replace(s, "&", "&amp;")
  s = Replace(s, "<", "&lt;")
  s = Replace(s, ">", "&gt;")
  s = Replace(s, """", "&quot;")
  HtmlEncode = s
End Function

Function HtmlAttrEncode(s)
  HtmlAttrEncode = HtmlEncode(s)
End Function

Sub OpenFolder(pathStr)
  shell.Run "explorer.exe """ & pathStr & """", 1, False
End Sub

Sub RunBat(folderPath)
  Dim batPath
  batPath = folderPath & "\" & BAT_NAME
  If fso.FileExists(batPath) Then
    shell.Run "cmd.exe /k ""cd /d """ & folderPath & """ & """ & BAT_NAME & """""", 1, True
  Else
    MsgBox "バッチが見つかりません：" & vbCrLf & batPath, 48, "エラー"
    LogError "BAT missing: " & batPath
  End If
End Sub

Sub OpenExcelRel(relPath)
  Dim filePath
  filePath = fso.BuildPath(baseDir, relPath)
  If fso.FileExists(filePath) Then
    shell.Run "cmd.exe /c start """" """ & filePath & """", 1, False
  Else
    MsgBox "Excel ファイルが存在しません：" & vbCrLf & filePath, 48, "エラー"
    LogError "Excel missing(rel): " & filePath
  End If
End Sub

Sub OpenExcelAbs(filePath)
  If fso.FileExists(filePath) Then
    shell.Run "cmd.exe /c start """" """ & filePath & """", 1, False
  Else
    MsgBox "Excel ファイルが存在しません：" & vbCrLf & filePath, 48, "エラー"
    LogError "Excel missing(abs): " & filePath
  End If
End Sub

Sub AddToolToMemory(id, title, subtitle, actionType, target, buttonText, pathLabel, openPath)
  Dim rec
  rec = IniEscape(title) & "|" & IniEscape(subtitle) & "|" & IniEscape(actionType) & "|" & IniEscape(target) & "|" & IniEscape(buttonText) & "|" & IniEscape(pathLabel) & "|" & IniEscape(openPath)
  tools(id) = rec
End Sub

Sub ParseToolRecord(rec, ByRef title, ByRef subtitle, ByRef actionType, ByRef target, ByRef buttonText, ByRef pathLabel, ByRef openPath)
  Dim parts
  parts = Split(rec, "|")
  title      = IniUnescape(parts(0))
  subtitle   = IniUnescape(parts(1))
  actionType = IniUnescape(parts(2))
  target     = IniUnescape(parts(3))
  buttonText = IniUnescape(parts(4))
  pathLabel  = IniUnescape(parts(5))
  openPath   = IniUnescape(parts(6))
End Sub

Function NextId()
  Dim i, maxId, k
  maxId = 0
  For Each k In tools.Keys
    If IsNumeric(k) Then
      i = CInt(k)
      If i > maxId Then maxId = i
    End If
  Next
  NextId = CStr(maxId + 1)
End Function

Function SortIds(keys)
  Dim arr(), i, j, tmp
  ReDim arr(UBound(keys))
  For i = 0 To UBound(keys)
    arr(i) = keys(i)
  Next
  For i = 0 To UBound(arr) - 1
    For j = i + 1 To UBound(arr)
      If CLng(arr(j)) < CLng(arr(i)) Then
        tmp = arr(i) : arr(i) = arr(j) : arr(j) = tmp
      End If
    Next
  Next
  SortIds = arr
End Function

Sub EnsureDefaultTools()
  If fso.FileExists(fso.BuildPath(baseDir, TOOLS_FILE)) Then Exit Sub
  AddToolToMemory "1", "一、TeraTerm & WinScp起動", "サーバ接続", "bat", "C:\Users\Desktop\TOOL\ServerTool", "起動", "C:\Users\Desktop\TOOL\ServerTool", "C:\Users\Desktop\TOOL\ServerTool"
  SaveTools
End Sub

Sub LoadTools()
  Dim p, ts, line, fields, id, rec
  tools.RemoveAll
  p = fso.BuildPath(baseDir, TOOLS_FILE)
  If Not fso.FileExists(p) Then EnsureDefaultTools
  If Not fso.FileExists(p) Then Exit Sub
  Set ts = fso.OpenTextFile(p, 1, False, -1)
  Do While Not ts.AtEndOfStream
    line = Trim(ts.ReadLine)
    If line <> "" Then
      fields = Split(line, vbTab)
      If UBound(fields) >= 1 Then
        id = fields(0)
        rec = fields(1)
        If id <> "" Then tools(id) = rec
      End If
    End If
  Loop
  ts.Close
End Sub

Sub SaveTools()
  Dim p, ts, k
  p = fso.BuildPath(baseDir, TOOLS_FILE)
  Set ts = fso.OpenTextFile(p, 2, True, -1)
  For Each k In SortIds(tools.Keys)
    ts.WriteLine k & vbTab & tools(k)
  Next
  ts.Close
End Sub

Sub Render()
  Dim html, k, title, subtitle, actionType, target, buttonText, pathLabel, openPath
  html = ""
  For Each k In SortIds(tools.Keys)
    Call ParseToolRecord(tools(k), title, subtitle, actionType, target, buttonText, pathLabel, openPath)
    html = html & "<div class='card' data-id='" & k & "'>" & vbCrLf
    html = html & "  <div class='title'>" & HtmlEncode(title) & "</div>" & vbCrLf
    If subtitle <> "" Then
      html = html & "  <div class='subtitle'>" & HtmlEncode(subtitle) & "</div>" & vbCrLf
    Else
      html = html & "  <div class='subtitle' style='display:none'></div>" & vbCrLf
    End If
    html = html & "  <div class='actions'>" & vbCrLf
    html = html & "    <input class='launch' type='button' value='" & HtmlAttrEncode(buttonText) & "' onclick='VBScript:RunTool(""" & k & """)'>" & vbCrLf
    html = html & "    <input class='danger' type='button' value='削除' onclick='VBScript:DeleteTool(""" & k & """)'>" & vbCrLf
    html = html & "  </div>" & vbCrLf
    html = html & "  <div class='path' onclick='VBScript:OpenToolPath(""" & k & """)'>📁 " & HtmlEncode(pathLabel) & "</div>" & vbCrLf
    html = html & "  <div class='note'>type=" & HtmlEncode(actionType) & "</div>" & vbCrLf
    html = html & "</div>" & vbCrLf
  Next
  document.getElementById("grid").innerHTML = html
End Sub

Sub RunTool(id)
  On Error Resume Next
  If Not tools.Exists(id) Then Exit Sub
  Dim title, subtitle, actionType, target, buttonText, pathLabel, openPath
  Call ParseToolRecord(tools(id), title, subtitle, actionType, target, buttonText, pathLabel, openPath)
  Select Case LCase(actionType)
    Case "bat" : Call RunBat(target)
    Case "excel_rel" : Call OpenExcelRel(target)
    Case "excel_abs" : Call OpenExcelAbs(target)
    Case "open_folder" : Call OpenFolder(target)
    Case Else : MsgBox "不明な actionType", 48, "エラー"
  End Select
  If Err.Number <> 0 Then LogError "RunTool error: " & Err.Description
  On Error GoTo 0
End Sub

Sub OpenToolPath(id)
  On Error Resume Next
  If Not tools.Exists(id) Then Exit Sub
  Dim title, subtitle, actionType, target, buttonText, pathLabel, openPath
  Call ParseToolRecord(tools(id), title, subtitle, actionType, target, buttonText, pathLabel, openPath)
  If openPath <> "" Then
    Call OpenFolder(openPath)
  Else
    If LCase(actionType) = "bat" Then
      Call OpenFolder(target)
    ElseIf LCase(actionType) = "excel_abs" Then
      Call OpenFolder(fso.GetParentFolderName(target))
    ElseIf LCase(actionType) = "excel_rel" Then
      Call OpenFolder(fso.GetParentFolderName(fso.BuildPath(baseDir, target)))
    End If
  End If
  On Error GoTo 0
End Sub

Sub DeleteTool(id)
  If Not tools.Exists(id) Then Exit Sub
  If MsgBox("削除しますか？", vbYesNo + vbQuestion, "確認") = vbYes Then
    tools.Remove id
    SaveTools
    Render
  End If
End Sub

Sub OpenAddPage()
  Dim p, ret

      SaveTools
      Render
    End If
  End If
End Sub

Sub OpenConfig()
  shell.Run "notepad.exe """ & fso.BuildPath(baseDir, TOOLS_FILE) & """", 1, False
End Sub

  Render
End Sub
